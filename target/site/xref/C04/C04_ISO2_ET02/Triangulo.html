3   /**
4    * Es una clase que crea un objeto "Triangulo" que dados unos lados y angulos
5    * determina el tipo de triangulo que es.
6    * 
7    * @author Javier Santos Sanz y Gonzalo Prieto de la Barreda
8    * @version 1.1
9    */
10  public class Triangulo {
11  	// Declaramos las variables para evitar los "números mágicos".
12  	static final double ANGULOTOTAL = 180.0;
13  	static final double ANGULORECTO = 90.0;
14  	static final double MARGENERROR = 0.1;
15  	// Dentro de la definición de requisitos, los triángulos pueden ser incorrectos
16  	// o imposibles donde de verdad sabremos si lo que hacen es correcto es en el
17  	// método a implementar.
18  	String tipos[] = new String[2];
19  	private double lados[] = new double[3];
20  	private double angulos[] = new double[3];
21  	// this.angulos[1]=Math.trunc(this.angulos[1]);
22  
23  	public Triangulo(double[] lados, double[] angulos) {
24  		// Para formar un triangulo importan las posiciones por lo que asumiremos:
25  		// lados[0] es el lado a, angulos[0] es alfa, lados[1] es el lado b,
26  		// angulos[1] es beta, lados[2] es el lado c, angulos[2] es ganma
27  		this.lados = lados.clone();
28  		this.angulos = angulos.clone();
29  		this.tipos = devolverTipo();
30  	}
31  
32  	public String[] devolverTipo() {
33  		// Primer valor para ver su tipo dependiendo de sus lados y el segundo depende
34  		// de sus ángulos.
35  		String tipos[] = new String[2];
36  		String trianguloValido = verificarTriangulo();
37  
38  		// Si no es válido el triángulo, no puede ser de un tipo
39  		if (trianguloValido == "Triángulo Válido") {
40  			tipos[0] = clasificarAngulo();
41  			tipos[1] = clasificarLado();
42  		} else {
43  			tipos[0] = "Triángulo imposible";
44  			tipos[1] = "Triángulo imposible";
45  		}
46  
47  		return tipos;
48  	}

50  	public String clasificarLado() {
51  		// Lanzar excepción si la raiz de los catetos al cuadrado es diferente de la
52  		// hipotenusa.
53  		String tipoLado = "";
54  		if (this.lados[0] == this.lados[1] && this.lados[0] == this.lados[2]) {
55  			tipoLado = "Equilátero";
56  		} else if (this.lados[0] == this.lados[1] || this.lados[0] == this.lados[2] || this.lados[1] == this.lados[2]) {
57  			tipoLado = "Isósceles";
58  		} else {
59  			tipoLado = "Escaleno";
60  		}
61  		return tipoLado;
62  	}

64  	public String clasificarAngulo() {
65  		String tipoAngulo = "";
66  		double anguloMayor = 0.0;
67  
68  		for (int i = 0; i < 3; i++) {
69  			if (this.angulos[i] > anguloMayor)
70  				anguloMayor = this.angulos[i];
71  		}
72  
73  		if (anguloMayor > ANGULORECTO + MARGENERROR) {
74  			tipoAngulo = "Obtusángulo";
75  		} else if (anguloMayor > ANGULORECTO - MARGENERROR) {
76  			tipoAngulo = "Recto";
77  		} else {
78  			tipoAngulo = "Acutángulo";
79  		}
80  		return tipoAngulo;
81  	}

83  	public String verificarTriangulo() {
84  		double total = 0;
85  
86  		try {
87  			for (int i = 0; i < 3; i++) {
88  				if (this.lados[i] <= 0)
89  					throw new LadoInvalidoException("Existe un lado que es menor o igual a 0.");
90  			}
91  		} catch (LadoInvalidoException e) {
92  			System.out.println(e.getMessage());
93  			return "Error Lado Invalido";
94  		}
95  		try {
96  			for (int i = 0; i < 3; i++) {
97  				total += this.angulos[i];
98  				if (this.angulos[i] <= 0)
99  					throw new AnguloInvalidoException("El ángulo es menor o igual a 0.");
100 			}
101 		} catch (AnguloInvalidoException e) {
102 			System.out.println(e.getMessage());
103 			return "Error Angulo Invalido";
104 		}
105 		try {
106 			if (total - ANGULOTOTAL > MARGENERROR || total - ANGULOTOTAL < -MARGENERROR)
107 				throw new SumaAnguloInvalidaException("El triángulo no suma 180º.");
108 		} catch (SumaAnguloInvalidaException e) {
109 			System.out.println(e.getMessage());
110 			return "Error Suma Angulos Invalidos";
111 		}
112 
113 		try {
114 			if (!(verificarTrianguloTeoremaSeno())) {
115 				throw new TrianguloImposibleException("El triángulo no cumple el teorema del seno");
116 			}
117 		} catch (TrianguloImposibleException e) {
118 			System.out.println(e.getMessage());
119 			return "Error Triángulo imposible";
120 		}
121 		return "Triángulo Válido";
122 	}

124 	public boolean verificarTrianguloTeoremaSeno() {
125 		// El teorema del seno es la mejor forma de verificar si un triángulo es posible
126 		// o no
127 		// Para poder hacerlo usando estimaciones es necesario sacar valores inferiores
128 		// y superiores
129 		// y ver si el valor está entre estos. El valor varía mucho en función de la
130 		// cercanía/lejanía
131 		// con respecto a pi/2 o 90º, por lo que el margen de error en algunas ocasiones
132 		// es demasiado
133 		// y en otras demasiado poco.
134 		double angulosRedondeados[] = new double[3];
135 		double constante[] = new double[3];
136 		double teoremaSenoExtremos[][] = new double[3][2];
137 		boolean valido = true;
138 
139 		// Primero truncamos el valor de los angulos a 2 decimales que es sobre lo que
140 		// trabajaremos
141 		for (int i = 0; i < 3; i++) {
142 			angulosRedondeados[i] = Math.round(this.angulos[i] * 100);
143 			angulosRedondeados[i] = angulosRedondeados[i] / 100;
144 		}
145 		// Calculamos el teorema del seno correspondiente a cada lado/ángulo
146 		for (int i = 0; i < 3; i++) {
147 			constante[i] = this.lados[i] / Math.sin(Math.toRadians(this.angulos[i]));
148 		}
149 		// Calculamos los valores extremos para poder estimar, como en 90º está el
150 		// extremo de la función
151 		// el menor/mayor cambiará en función del grado
152 		for (int i = 0; i < 3; i++) {
153 			if (angulosRedondeados[i] >= 90.00) {
154 				if ((angulosRedondeados[i] - MARGENERROR) < 90.00) {
155 					teoremaSenoExtremos[i][0] = lados[i] - MARGENERROR;
156 				} else {
157 					teoremaSenoExtremos[i][0] = this.lados[i]
158 							/ Math.sin(Math.toRadians(angulosRedondeados[i]) - MARGENERROR);
159 				}
160 				teoremaSenoExtremos[i][1] = this.lados[i]
161 						/ Math.sin(Math.toRadians(angulosRedondeados[i]) + MARGENERROR);
162 			} else {
163 				if ((angulosRedondeados[i] + MARGENERROR) > 90.00) {
164 					teoremaSenoExtremos[i][0] = lados[i] - MARGENERROR;
165 				} else {
166 					teoremaSenoExtremos[i][0] = this.lados[i]
167 							/ Math.sin(Math.toRadians(angulosRedondeados[i]) + MARGENERROR);
168 				}
169 				teoremaSenoExtremos[i][1] = this.lados[i]
170 						/ Math.sin(Math.toRadians(angulosRedondeados[i]) - MARGENERROR);
171 			}
172 		}
173 		// Finalmente, se realiza la comparación, en caso de que algún valor no coincida
174 		// se evaluará en falso
175 		// También se usa el calcular el módulo junto con sumar 1 y 2 para poder
176 		// aplicarlo sin tener en cuenta el valor de i.
177 		for (int i = 0; i < 3; i++) {
178 			if (!(((constante[i] > teoremaSenoExtremos[(i + 1) % 3][0])
179 					&& (constante[i] > teoremaSenoExtremos[(i + 2) % 3][0]))
180 					&& ((constante[i] < teoremaSenoExtremos[(i + 1) % 3][1])
181 							&& (constante[i] < teoremaSenoExtremos[(i + 2) % 3][1])))) {
182 				valido = false;
183 			}
184 		}
185 		return valido;
186 	}
